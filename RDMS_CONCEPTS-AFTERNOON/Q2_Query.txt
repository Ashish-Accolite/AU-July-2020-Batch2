SELECT EMP.NAME AS DOC_NAME, EMP_ID AS DOC_ID, EMP_TYPE, DEPT, PATIENT.PID AS PATIENT_ID, PATIENT.NAME AS PATIENT_NAME FROM (SELECT * FROM EMPLOYEE WHERE EMPLOYEE.PROFESSION = "Doctor") AS EMP LEFT OUTER JOIN RECORD AS REC ON EMP.EMP_ID = REC.FK_RECORD_DOC_ID  LEFT OUTER JOIN PATIENT ON REC.FK_RECORD_PID = PATIENT.PID;
